数据存储

发表于 2018-08-27

目的

当业务大的时候，数据量就会很大，所以需要数据存储，和查询

数据储存的方式

  1. txt
  2. json
  3. csv
  4. 关系型数据库： Mysql
  5. 非关系型数据库： Mongodb

实践操作

  1. txt

    # 写
data = '我是谁'
with open('test.txt', 'w', encoding="utf-8") as fp:
    fp.write(data)
# 多行写
datalist = [
    '我是谁',
    '我要上学校',
    '天天不迟到',
    '哈哈',
]
with open('test.txt', 'w', encoding="utf-8") as fp:
    for data in datalist:
        dataline = data + '\n'
        fp.write(dataline)

# 读
with open('test.txt', encoding="utf-8") as fp:
      data = fp.read()

# 多行读
with open('test.txt', encoding="utf-8") as fp:
      for line in fp.readlines():
          print(line.strip())

  2. json

    # 写
data_json = [
  {
      'user' : 'ming',
      'password' : 'ming;;;;'
  },
  {
      'user' : 'tao',
      'password' : '1234'
  }
  ]
with open('test.json', 'w') as fp:
      data_str = json.dumps(data_json, indent=2)   # 将json数据转化成字符串
      fp.write(data_str)

# 读
def jsonread():
    with open('test.json') as fp:
        data_str = fp.read()
        data_json = json.loads(data_str)
    print(data_json)

  3. csv

    import csv
with open('test.csv', 'w') as fp:
    writer = csv.writer(fp)             # csv 需要使用自己的write方法
    writer.writerow(['id', 'name', 'age'])
    writer.writerow(['1001', 'Mike', '20'])
    writer.writerow(['1002', 'Bob', '22'])
    writer.writerow(['1003', 'Jordan', '21'])

# 字典方式写
with open('test.csv', 'w') as fp:
    fieldnames = ['id', 'name', 'age']
    writer = csv.DictWriter(fp, fieldnames=fieldnames)
    writer.writeheader()
    writer.writerow({'id':'1001', 'name': 'Mike', 'age': 20})
    writer.writerow({'id':'1002', 'name': 'Bob', 'age': 22})
    writer.writerow({'id':'1003', 'name': 'Jordan', 'age': 21})

# 读
with open('test.csv') as fp:
    reader = csv.reader(fp)
    for row in reader:
        print(row)

    # 连接
conn = pymysql.connect(host='localhost', user='root', password='root', port=3306)
cursor = conn.cursor()

# 执行
sql = ''

# 关闭
cursor.close()

# 字典插入方式
data = {
      'id' : '003',
      'name' : 'BOb',
      'age': 22
  }
  table = 'testtable'
    keys = ', '.join(data.keys())
    values = ', '.join(['%s'] * len(data))
    sql = 'insert into {table} ({keys}) values ({values})'.format(table=table, keys=keys, values=values)
    try:
        if cursor.execute(sql, tuple(data.values())):
            print('Successful')
            db.commit()
    except:
        print('Failed')
        db.rollback()

# 跟新操作
sql = 'update testtable set age = %s where name = %s'
  try:
      cursor.execute(sql, (25, 'Bob'))
      db.commit()
  except:
      db.rollback()

# 删除操作
table = 'testtable'
  condition = 'age > 20'
  sql = 'delete from {table} where {condition}'.format(table=table, condition=condition)
  try:
      cursor.execute(sql)
      db.commit()
  except:
      db.rollback()

# 查询操作
sql = 'select  * from testtable Where age >= 20'
try:
    cursor.execute(sql)
    print('Count:', cursor.rowcount)
    row = cursor.fetchone()
    while row:
        print(row)
        row = cursor.fetchone()
except:
    print('Error')
