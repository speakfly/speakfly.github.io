目的

了解Selenium 模拟浏览器运行的方式实现

实践操作

  1. Selenium打开浏览器并访问页面

    # 打开各种浏览器
from selenium import webdriver

browser = webdriver.Chrome()
browser = webdriver.Firefox()
browser = webdriver.Edge()
browser = webdriver.PhantomJS()
browser = webdriver.Safari()

#访问页面
browser = webdriver.Firefox()
browser.get('https://www.taobao.com')
print(browser.page_source)
browser.close()

  2. 获取节点

    #获取节点
input_first = browser.find_element_by_id('q')

input_third = browser.find_element_by_xpath('//*[@id="q"]')

  4. 动作链

    from selenium import webdriver
from selenium.webdriver import ActionChains

browser = webdriver.Firefox()
url = "http://wwww.runoob.com/try/try.php?filename=jqueryui-api-drappable"
browser.get(url)
browser.switch_to.frame('i')

source = browser.find_element(By.CSS_SELECTOR, '#draggable')
target = browser.find_element(By.CSS_SELECTOR, '#droppable')
actions = ActionChains(browser)       #主要是这三句
actions.drag_and_drop(source, target)
actions.perform()

  8. 延时等待

    #访问网页的时候，我们需要等网页加载好了才能进行下一步的处理
#那么这涉及到我们要等到什么时候才知道加载完成

# 隐式等待 使用查找节点的方法 如果超时了就证明加载没完成

from selenium import webdriver

brower = webdriver.Firefox()
brower.implicitly_wait(10) #默认值为0
brower.get('https://www.zhihu.com/explore')
input = brower.find_element(By.CLASS, 'zu-top-add-question')
print(input)

# 显式等待
# 我们比较常用的是显式等待  同样的原理但是如果找不到节点会抛出异常

from selenium import webdriver
from selenium.webdriver.common.by import By
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.support.wait import WebDriverWait

brower = webdriver.Firefox()
brower.get('https://www.taobao.com')
wait = WebDriverWait(brower, 10)   #主要是这两句
input = wait.until(EC.presence_of_element_located((By.ID, 'q')))
button = wait.until(EC.element_to_be_clickable(By.CSS_SELECTOR, '.btn-search')))
print(input, button)

# 还有其他等待条件 我就不一一表明

  12. 异常管理

    from selenium import webdriver
from selenium.common.exceptions import TimeoutException, NoSuchElementException

brower = webdriver.Firefox()
try:
  brows.get('https://www.baidu.com')
except TimeoutException:

except NoSuchElementException:

finally:
  browser.close()
